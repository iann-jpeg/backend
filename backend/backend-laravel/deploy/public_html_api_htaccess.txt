# Sample .htaccess for public_html/api/
<IfModule mod_rewrite.c>
    RewriteEngine On
    RewriteBase /api/

    # Redirect all requests to index.php
    RewriteCond %{REQUEST_FILENAME} !-f
    RewriteCond %{REQUEST_FILENAME} !-d
    RewriteRule ^ index.php [L]
</IfModule>
